Aiohttp调用Http代理
Python3使用aiohttp调用Http代理IP的代码样例
温馨提示：
1.
aiohttp的代码样例支持访问http,https网页
2.

3.

使用代码示例


import asyncio
import aiohttp
import requests

# python的3.13.5以下版本如果不可用可按照下面的方案修改
# aiohttp目录下找到Lib\site-packages\aiohttp\ 找到 connector.py然后# 注释如下行

# # 新增
# runtime_has_start_tls = False if req.proxy.scheme != "https" else self._loop_supports_start_tls()


async def fetch(session, url):
    
    headers = {
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/124.0.0.0 Safari/537.36'
    }
    #账号密码认证模式  
    #username = "username"
    #password = "password"
    #proxy_auth = aiohttp.BasicAuth(username, password)
    #proxy = "http://ip:port" 
    #async with session.get(url, headers=headers, ssl=False, proxy=proxy, proxy_auth=proxy_auth, timeout=10) as response:
    
    #白名单模式  
    proxy = "http://51.79.78.230:10011"
    async with session.get(url, headers=headers, ssl=False, proxy=proxy, timeout=10) as response:
        return await response.text(), response.status


async def main():
    try:
        async with aiohttp.ClientSession(connector=aiohttp.TCPConnector(ssl=False), trust_env=True) as session:
            content, status = await fetch(session, 'https://myip.ipip.net/')
            print(f"status: {status}")
            print(f"content: {content}")
    except aiohttp.client_exceptions.ClientConnectorError as e:
        print(f'连接代理或目标服务器时发生错误: {e}')


if __name__ == '__main__':
    policy = asyncio.WindowsSelectorEventLoopPolicy()
    asyncio.set_event_loop_policy(policy)
    asyncio.run(main())

import http.client
import ssl


# 自定义 SSL 上下文
ssl_context = ssl.create_default_context()
ssl_context.minimum_version = ssl.TLSVersion.TLSv1_2
ssl_context.check_hostname = False
ssl_context.verify_mode = ssl.CERT_NONE  # 不验证证书（测试用）

# 代理服务器的地址和端口
proxy_host = "36.134.55.149:25577"

headers = {
    "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/58.0.3029.110 Safari/537.36"
}

# 目标服务器的地址
target_host = "https://myip.ipip.net/"

# 创建连接对象并指定 SSL 上下文
conn = http.client.HTTPSConnection(proxy_host, context=ssl_context)

# 设置代理隧道
conn.set_tunnel(target_host)

# 发送请求
conn.request("GET", "/", headers=headers)

# 获取响应
response = conn.getresponse()
print(response.status, response.reason)
print(response.read().decode('utf-8'))

# 关闭连接
conn.close()



Websocket调用HTTP代理
Python3使用websocket调用Http代理IP的代码样例
温馨提示：
1.
安装运行所需的客户端: pip install websocket-client
2.
使用HTTP代理发送websocket请求
3.
在IP可用的情况下，客户端长时间不发送消息，服务端会断开连接
4.
运行环境要求 python3.x

代码样例中的代理IP和端口均为虚构，请通过seekproxy的API提取链接获取代理IP使用，避免报错
使用代码示例


#!/usr/bin/env python
# -*- encoding: utf-8 -*-
import ssl
import websocket


def on_message(ws, message):
    print('响应数据:', message)


def on_error(ws, error):
    print(error)


def on_open(ws):
    data = '{}'  # 此处填入您需要传给目标网站的json格式参数，如{"type":"web","data":{"_id":"xxxx"}}
    ws.send(data)


def on_close(*args):
    print("### closed ###")


def start():
    websocket.enableTrace(True)
    target_url = 'ws://myip.ipip.net'  # 此处替换您的目标网站
    ws = websocket.WebSocketApp(
        url=target_url,
        header=[
            "User-Agent: Mozilla/5.0 (Macintosh; Intel Mac OS X 10_15_1) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/85.0.4183.102 Safari/537.36"
        ],
        on_message=on_message,
        on_error=on_error,
        on_close=on_close,
    )
    ws.on_open = on_open
    ws.run_forever(
        sslopt={"cert_reqs": ssl.CERT_NONE},
        http_proxy_host="103.43.135.4",
        http_proxy_port=45815,
        proxy_type="http",
    )


if __name__ == "__main__":
    start()
